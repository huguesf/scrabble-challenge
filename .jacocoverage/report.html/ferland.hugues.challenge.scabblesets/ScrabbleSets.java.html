package ferland.hugues.challenge.scabblesets;

import java.util.stream.Collectors;
import static java.util.stream.Collectors.collectingAndThen;
import static java.util.stream.Collectors.joining;
import static java.util.stream.Collectors.mapping;
import static java.util.stream.Collectors.toList;

public class ScrabbleSets {

    private final static String ALPHABET = "ABCDEFGHIJKLMNOPQRSTUVWXYZ_";

    public static void main(String[] args) {
        Bag bag = new Bag();
        
        try {
            args[0].chars().forEach(c -> bag.remove(c));
            
            ALPHABET
                    .chars()
                    .mapToObj(o -> (char)o)
                    .collect(Collectors.groupingBy((c) -> bag.count(c)))
                    .entrySet()
                    .stream()
                    .sorted((e1, e2) -> e2.getKey() - e1.getKey())
                    .forEach((e) -> {
                        System.out.println(e.getKey() + ": " + e.getValue().stream().sorted().map(o -> o.toString()).collect(joining(", ")));
                            });
        } catch (IllegalStateException e) {
            System.out.println(e.getMessage());
        }
    }
}
